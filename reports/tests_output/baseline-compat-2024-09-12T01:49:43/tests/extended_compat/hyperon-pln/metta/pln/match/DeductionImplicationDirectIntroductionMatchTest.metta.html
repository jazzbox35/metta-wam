;;;;

;;;; Recursive step (inductive rule):
;;;;
;;;; (p a) ≞ patv
;;;; (q a) ≞ qatv
;;;; p→q ≞ (ETV ev pqtv)
;;;; a ∉ ev
;;;; ⊢
;;;; p→q ≞ (ETV (insert a ev) tv)
;;;;
;;;; where tv is calculated as defined in the idi_formula.
;;;;
;;;; IDI rule as a function

    (idi_induction 
      (≞

        ($q $a) $qatv)
